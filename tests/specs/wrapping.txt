== short command stays single line ==
message(hello world)

[expect]
message(hello world)

== long command wraps to multi-line ==
target_link_libraries(my_target public libfoo libbar libbaz libanother_lib libyet_another)
[expect]
target_link_libraries(
  my_target
  PUBLIC
  libfoo
  libbar
  libbaz
  libanother_lib
  libyet_another
)

== moderate command wraps at narrow line width ==
set(MY_VARIABLE value_one value_two xyzwq)
[expect]
set(MY_VARIABLE value_one value_two xyzwq)

== many arguments each on own line ==
list(APPEND MY_LIST alpha beta gamma delta epsilon zeta eta theta iota kappa)
[expect]
list(APPEND MY_LIST alpha beta gamma delta epsilon zeta eta theta iota kappa)

== closing paren on last argument line ==
target_link_libraries(my_target PUBLIC libfoo libbar libbaz libanother_lib libyet_another)
[expect]
target_link_libraries(
  my_target
  PUBLIC
  libfoo
  libbar
  libbaz
  libanother_lib
  libyet_another
)
